# Function to perform Augmented Dickey-Fuller test
def adf_test(series, title=''):
    print(f'\nADF Test for {title}:')
    result = adfuller(series.dropna(), autolag='AIC')
    print(f'ADF Statistic: {result[0]:.4f}')
    print(f'p-value: {result[1]:.4f}')
    print(f'Critical Values:')
    for key, value in result[4].items():
        print(f'   {key}: {value:.4f}')
    if result[1] <= 0.05:
        print(f"Conclusion: Reject the null hypothesis - {title} is stationary.")
    else:
        print(f"Conclusion: Fail to reject the null hypothesis - {title} is non-stationary.")

# Perform ADF test on GS3M and GS6M
adf_test(df['GS3M'], 'GS3M')
adf_test(df['GS6M'], 'GS6M')

# Test for cointegration using Engle-Granger method
print("\nCointegration Test (Engle-Granger):")
coint_test = coint(df['GS3M'], df['GS6M'])
print(f'Cointegration t-statistic: {coint_test[0]:.4f}')
print(f'p-value: {coint_test[1]:.4f}')
print(f'Critical Values: {coint_test[2]}')
if coint_test[1] <= 0.05:
    print("Conclusion: Reject the null hypothesis - GS3M and GS6M are cointegrated.")
else:
    print("Conclusion: Fail to reject the null hypothesis - GS3M and GS6M are not cointegrated.")
