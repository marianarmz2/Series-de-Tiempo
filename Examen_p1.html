# Convert to DataFrame
df = pd.DataFrame(data)

# Fix the YEAR column by splitting and converting to datetime
df['YEAR'] = df['YEAR'].apply(lambda x: f"{x.split('.')[0]}-Q{x.split('.')[1]}")
df['YEAR'] = pd.to_datetime(df['YEAR'].replace('Q1', '01-01').replace('Q2', '04-01')
                             .replace('Q3', '07-01').replace('Q4', '10-01'))
df.set_index('YEAR', inplace=True)

# Step 2: Apply log transformation to PDI
df['Log_PDI'] = np.log(df['PDI'])

# Step 3: Visualize the data
plt.figure(figsize=(10, 6))
plt.plot(df['Log_PDI'], label='Log PDI')
plt.title('Log Personal Disposable Income (1970.1 - 1991.4)')
plt.xlabel('Year')
plt.ylabel('Log PDI')
plt.legend()
plt.show()

# Step 4: Check stationarity with ADF test
def adf_test(series, title=''):
    result = adfuller(series.dropna())
    print(f'ADF Test on {title}:')
    print('ADF Statistic:', result[0])
    print('p-value:', result[1])
    print('Critical Values:', result[4])
    if result[1] <= 0.05:
        print("The series is stationary\n")
    else:
        print("The series is non-stationary\n")

adf_test(df['Log_PDI'], 'Log PDI')

# Step 5: Difference the series and recheck stationarity
df['Diff_Log_PDI'] = df['Log_PDI'].diff()
adf_test(df['Diff_Log_PDI'], 'Differenced Log PDI')

# Step 6: Plot ACF and PACF for differenced series
plt.figure(figsize=(12, 6))
plt.subplot(211)
plot_acf(df['Diff_Log_PDI'].dropna(), ax=plt.gca(), lags=20)
plt.title('ACF of Differenced Log PDI')
plt.subplot(212)
plot_pacf(df['Diff_Log_PDI'].dropna(), ax=plt.gca(), lags=20)
plt.title('PACF of Differenced Log PDI')
plt.tight_layout()
plt.show()

# Step 7: Fit ARIMA model (initial guess based on ACF/PACF)
# Try ARIMA(1,1,1) as a starting point, then refine
p, d, q = 1, 1, 1  # Adjust based on ACF/PACF analysis
model = ARIMA(df['Log_PDI'], order=(p, d, q))
results = model.fit()
print(results.summary())

# Step 8: Diagnostics - Plot residuals
residuals = results.resid
plt.figure(figsize=(10, 6))
plt.plot(residuals)
plt.title('Residuals of ARIMA(1,1,1) Model')
plt.xlabel('Time')
plt.ylabel('Residuals')
plt.show()

# ACF of residuals
plt.figure(figsize=(10, 4))
plot_acf(residuals, lags=20)
plt.title('ACF of Residuals')
plt.show()

# Step 9: Forecast the next 4 quarters
forecast = results.forecast(steps=4)
forecast_index = pd.date_range(start=df.index[-1], periods=5, freq='Q')[1:]
forecast_series = pd.Series(forecast, index=forecast_index)

# Plot forecast
plt.figure(figsize=(10, 6))
plt.plot(df['Log_PDI'], label='Observed Log PDI')
plt.plot(forecast_series, label=f'Forecasted Log PDI (ARIMA{p,d,q})', color='red')
plt.title(f'ARIMA({p},{d},{q}) Forecast of Log PDI')
plt.xlabel('Year')
plt.ylabel('Log PDI')
plt.legend()
plt.show()

# Convert forecast back to original scale
forecast_original_scale = np.exp(forecast)
print("\nForecasted PDI (Original Scale) for next 4 quarters:")
for date, value in zip(forecast_index, forecast_original_scale):
    print(f"{date}: {value:.2f}")

# Step 10: Model Selection (optional manual tuning)
# Try a few other (p,d,q) combinations based on ACF/PACF and AIC/BIC
orders_to_try = [(1, 1, 0), (0, 1, 1), (2, 1, 1), (1, 1, 2)]
for order in orders_to_try:
    model = ARIMA(df['Log_PDI'], order=order)
    results = model.fit()
    print(f"\nARIMA{order} AIC: {results.aic}")
    print(f"ARIMA{order} BIC: {results.bic}")
